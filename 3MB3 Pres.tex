\section{Arms Race Model}

\begin{frame}{Arms Race Model}
    \begin{block}{Model Equations}
        \begin{align*}
        \frac{\partial p}{\partial t} &= a g(t) - b p(t) + r \\\\
        \frac{\partial g}{\partial t} &= c p(t) - d g(t) + s
    \end{align*}
    \end{block}
    \begin{itemize}
        \item a and c: Mutual Fear Factor (\(a,c>0\))
        \item b and d: Expenditure Factor (\(b,d>0\))
        \item r and s: Goodwill/Grievance Factor ((\(r,s<0\)) for goodwill, 
        (\(r,s>0\)) for grievance)
    \end{itemize}
\end{frame}
