\section{Overview}
\begin{frame}{Overview}
	% \textbf{Scenario}: Purple (p) and Green (g) are 2 countries competing in an arms race where each one is:
    % \begin{itemize}

    \textbf{Scenario}: Purple (p) and Green (g) are two countries competing in an arms race. Each is:
    \begin{itemize}
        \item More inclined to increase defence spending as the other country's spending rises
        \item Less inclined to increase spending as its own spending rises
    \end{itemize}

    \textbf{Research Question}: What happens to the long-term spending of each country? 

    \textbf{Key Factors}:
    \begin{itemize}
        \item How do goodwill and fear between countries affect defence spending?
        \item How do spending caps affect a country's position in the arms race?
    \end{itemize}

    % \textbf{Model Overview}:
    % \begin{itemize}
    %     \item $\Delta$spending = x(other country's spending) - y(own spending)
    %     \item x and y measure the severity of encouragement or discouragement to spend
    % \end{itemize}
\end{frame}
% Section
\section{Base Model}
\begin{frame}{Arms Race Base Model}
    \begin{block}{Model Equations}
        \begin{align*}
        \frac{\partial p}{\partial t} &= a g(t) - b p(t) \\
        \frac{\partial g}{\partial t} &= c p(t) - d g(t)
        \end{align*}
    \end{block}
    \begin{itemize}
        \item Feasibility Condition: \((p(t),g(t) \ge 0)\)
        \item \(a,c\): Mutual Fear Factor (\(a,c>0\))
        \item \(b,d\): Expenditure Factor (\(b,d>0\))
    \end{itemize}
\end{frame}

\section{Grievances and Goodwill}
\begin{frame}{Grievances and Goodwill}
    Suppose countries p and g have underlying goodwill/grievances that influence their spending \\
    The model equations are therefore:
    \begin{align*}
        \frac{\partial p}{\partial t} &= a g(t) - b p(t) + r\\
        \frac{\partial g}{\partial t} &= c p(t) - d g(t) + s
